<script type="text/javascript">
/* Lorsque le texte situé dans la case de l'événement est trop long, il est coupé à
     partir du bord de la case. Cette fonction agrandit la case en hauteur et en largeur
     jusqu'à ce que le texte entier soit lisible.

     Lors de l'agrandissement, la largeur et la hauteur de la case ne doivent pas devenir plus petit
     que la case dans sa taille initiale. Car :
       - lorsque cette fonction agrandit la case et que la hauteur devient plus courte qu'initialement, le
         curseur peut se retrouver en dehors de la case et déclencher un événement "onmouseout". Cet événement
         va appeler la fonction "reduireCase" qui va redonner la taille initiale de la case. Sans avoir bougé,
         le curseur se retrouve de nouveau dans la case, ce qui génère un événement "onmouseover". Et ainsi de
         suite. Créer un système pour éviter que cela parte en boucle est plus complexe que d'agrandir la case
         de façon à ce que la hauteur et la largeur soient plus grand que ceux de la case initiale,
       - même si la case garde son point d'origine (en haut à gauche) au même endroit, le fait qu'elle
         ait une taille différente rend plus difficile la lecture du planning.
*/
function agrandirCase(element, event) {
    // Si la touche contrôle est maintenue appuyée pendant le survol du curseur, cette fonction est désactivée
    if (event.ctrlKey == true)
        return;
    
    var MAX_SCROLL = 1000000;
    var INCREMENT = 50;
    var je = jQuery(element); // je : jQuery Element

    // tente de scroller verticalement et horizontalement au maximum
    je.scrollTop(MAX_SCROLL);
    je.scrollLeft(MAX_SCROLL);

    // le div a défilé, ce qui indique qu'une partie du texte n'est pas lisible
    if (je.scrollTop() != 0 || je.scrollLeft() != 0) {
        je.addClass("evenementAgrandi");

        // sauvegarde de la taille actuelle pour la restaurer lors de "reduireCase"
        element.oldWidth = je.width();
        element.oldHeight = je.height();

        var maxWindowHeight = jQuery(window).height();
        var maxWindowWidth = jQuery(window).width();

        // Agrandit la largeur du div à la largeur du mot le plus long
        // tant que le div peut défiler et que la largeur du div ne dépasse pas la largeur de la fenêtre
        while (je.scrollLeft() != 0 && je.width() + INCREMENT <= maxWindowWidth) {
            je.width(je.width() + INCREMENT);
            je.scrollLeft(MAX_SCROLL);
        }

        // Agrandit la hauteur du div
        while (je.scrollTop() != 0 && je.height() + INCREMENT <= maxWindowHeight) {
            je.height(je.height() + INCREMENT);
            je.scrollTop(MAX_SCROLL);
        }

        // Dans le cas où les barres de défilement seraient encore affichées, repositionne le texte au début
        je.scrollTop(0);
        je.scrollLeft(0);
    }
}

// Rétablit la taille initiale de la case de l'événement
function reduireCase(element, event) {
    var je = jQuery(element); // je : jQuery Element

    je.width(element.oldWidth);
    je.height(element.oldHeight);
    je.removeClass("evenementAgrandi");
    
    // repositionne le texte au début
    je.scrollTop(0);
    je.scrollLeft(0);
}
